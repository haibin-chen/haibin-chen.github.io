Standford Corenlp工具包功能文档

一、安装

（1）java包解压，项目构建可参考https://github.com/stanfordnlp/CoreNLP使用ant或maven构建，电脑需要安装java 1.8以上版本。
（2）官方有封装好的python包stanfordcorenlp，使用pip install stanfordcorenlp安装
注：若需要其他语言如中文需要另外下载语言包放到文件夹中。

import requests
import json

sentence = ‘An exploration vessel performing deepwater seismic activities in the Kara Sea, off the coast of Russia.’
sentence = sentence.replace(“ “, “ “).encode(‘utf-8’)
print(“process sentence: “, sentence)
process_str = requests.post(‘http://localhost:9000/?properties={"annotators":"tokenize, ssplit, pos, parse”,”outputFormat”:”json”}’, data=sentence).text
#print(process_str)
process_json = json.loads(process_str)

print()
print(“Part 1: part-of-speech result word by word:”)
for i in range(len(process_json[‘sentences’])):
print(“sentence “, i+1)
token_list = process_json[‘sentences’][i][‘tokens’]
for token in token_list:
print(token[‘word’] + “ “ + token[‘pos’])

print()
print(“Part 2: constituency parsing in tree format:”)
for i in range(len(process_json[‘sentences’])):
print(“sentence “, i+1)
print(process_json[‘sentences’][i][‘parse’])
print()
print(“Part 3: constituency parsing in sequence format(remove new line and blank space):”)
for i in range(len(process_json[‘sentences’])):
print(“sentence “, i+1)
print(process_json[‘sentences’][i][‘parse’].replace(“\n”,””).replace(“ “,””))

class Node:
parent = “”
child = []
value = “”

def construct_parsing_tree(s):
s = s[1:-1]
root = 0
i = 0
lastNode = 0
nodeStack = []
while i < len(s):
if s[i] == ‘(‘:
i += 1
elif s[i] == ‘)’:
nodeStack.pop()
i += 1
elif s[i] == ‘ ‘:
i += 1
else:
word = “”
while i < len(s) and s[i] != ‘(‘ and s[i] != ‘)’:
word += s[i]
i += 1
lastNode = Node()
lastNode.value = word
lastNode.child = []
if root == 0:
root = lastNode
else:
parent_node = nodeStack[-1]
parent_node.child.append(lastNode)
nodeStack.append(lastNode)
return root

print()
print(“Part 4: constituency parsing by tree node(from top to bottom):”)
for i in range(len(process_json[‘sentences’])):
print(“sentence “, i+1)
process_tree = construct_parsing_tree(process_json[‘sentences’][i][‘parse’].replace(“\n”,””).replace(“ “,””))
currentStack = [process_tree]
while currentStack:
nextStack = []
for tree in currentStack:
if tree.child != []:
nextStack.extend([subtree for subtree in tree.child])
tree_child = [subtree.value for subtree in tree.child]
print(“tree node value:” + str(tree.value) + “ tree node child:” + str(tree_child))
currentStack = nextStack
print()
print(“process finish”)
